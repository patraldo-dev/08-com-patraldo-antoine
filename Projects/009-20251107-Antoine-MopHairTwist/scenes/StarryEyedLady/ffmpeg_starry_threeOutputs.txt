#!/bin/bash

VIDEO="StarryEyed_wXfades.mp4"
MUSIC="audio/TheTwist.m4a"
DIALOGUE1="audio/star_en.mp3"
DIALOGUE2="audio/star_es.mp3"
DIALOGUE3="audio/star_fr.mp3"

# Volume envelope for music: loud during long (10s) clips, soft during short (5s) clips
MUSIC_VOLUME_EXPR="if(between(t,0,10),1,if(between(t,10,15),0.2,if(between(t,15,25),1,if(between(t,25,30),0.2,if(between(t,30,40),1,0.2)))))"

for i in 1 2 3; do
  DIALOGUE_VAR="DIALOGUE${i}"
  DIALOGUE="${!DIALOGUE_VAR}"
  OUTPUT="output_lang${i}.mp4"

  ffmpeg -i "$VIDEO" -i "$DIALOGUE" -i "$MUSIC" -filter_complex "
    [1:a]apad=pad_dur=60,asetpts=N/SR/TB[dlg];
    [2:a]volume=$MUSIC_VOLUME_EXPR[music_vol];
    [dlg][music_vol]amix=inputs=2:duration=longest[aout]
  " -map 0:v -map "[aout]" -c:v copy -c:a aac -shortest "$OUTPUT"

  echo "Created $OUTPUT"
done
